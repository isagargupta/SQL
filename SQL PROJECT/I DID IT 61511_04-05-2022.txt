CREATE VIEW [dbo].[NAME_1]
AS
SELECT dbo.data_person.id_person, dbo.name.name
FROM     dbo.data_person INNER JOIN
                  dbo.name ON dbo.data_person.id_name = dbo.name.id_name
WHERE  (dbo.data_person.id_Name_of_name = 1)

CREATE VIEW [dbo].[NAME_SURNAME_OF_PERSON]
AS
SELECT dbo.person.BirthDate, dbo.NAME_1.name AS NAME_1, dbo.SURNAME.name AS SURNAME
FROM     dbo.person INNER JOIN
                  dbo.NAME_1 ON dbo.person.id_person = dbo.NAME_1.id_person INNER JOIN
                  dbo.SURNAME ON dbo.person.id_person = dbo.SURNAME.id_person

CREATE PROCEDURE [dbo].[ADD_DATA]
@ID_PERSON INT,
@NAME VARCHAR(40),
@id_Name_of_name INT
AS
BEGIN
declare @id_name INT
exec is_name @NAME, @id_name output
INSERT INTO DATA_PERSON VALUES (@ID_PERSON,@id_name,@id_Name_of_name)
END

CREATE PROCEDURE [dbo].[ADD_PERSON]
@DATE_OF_BIRTH DATE,
@NAME_1 VARCHAR (40),
@SURNAME VARCHAR (40)
AS
BEGIN
DECLARE @ID_PERSON INT
DECLARE @id_name INT
INSERT INTO PERSON VALUES(@DATE_OF_BIRTH)
SET @ID_PERSON=@@IDENTITY

exec ADD_DATA @ID_PERSON,@NAME_1,1
exec ADD_DATA @ID_PERSON,@SURNAME,3

END
GO

CREATE PROCEDURE [dbo].[is_name]
@name varchar (40),
@id_name int output
AS
BEGIN
select @id_name=id_name from [name] where [name]=@name
if @id_name is null
begin
insert into [name] values (@name)
set @id_name=@@identity
end
END
