CREATE DATABASE Problem1;
use Problem1;


CREATE TABLE GENDER(
    ID_GENDER INT(10) PRIMARY KEY,
    NAME VARCHAR(20)
);


CREATE TABLE PERSON(
    ID_PERSON INT(10) PRIMARY KEY,
    NAME VARCHAR(20),
    SURNAME VARCHAR(40),
    ID_GENDER INT(10),
    FOREIGN KEY(ID_GENDER) REFERENCES GENDER(ID_GENDER)
);

CREATE TABLE EMAIL(
    ID_EMAIL INT(10) PRIMARY KEY,
    EMAIL VARCHAR(60),
    ID_PERSON INT(10),
    FOREIGN KEY(ID_PERSON) REFERENCES PERSON(ID_PERSON)
);
